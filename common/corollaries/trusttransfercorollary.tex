\begin{corollary}[Trust Transfer Corollary] \ \\
  \label{trusttransfer}
  Let $V \in \left[0, F\right]$. We create a $C'$ where
  \begin{equation*}
  \begin{gathered}
    \forall i \in [n], c_i' \leq x_i \mbox{ and}\\
    \sum\limits_{i=1}^{n}{c_i'} = F - V \enspace.
  \end{gathered}
  \end{equation*}

  It then holds that $F' = F - V$.
\end{corollary}
